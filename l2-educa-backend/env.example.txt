# ================================================
# L2 EDUCA BACKEND - ENVIRONMENT CONFIGURATION
# ================================================
# Copy this file to .env and fill in your actual values
# NEVER commit the .env file to git!
# ================================================

# === Server Configuration ===
NODE_ENV=production
PORT=3001

# === JWT Configuration ===
# Generate a strong secret (min 32 characters):
# Windows PowerShell: -join ((65..90) + (97..122) + (48..57) | Get-Random -Count 32 | % {[char]$_})
# Mac/Linux: openssl rand -base64 32
JWT_SECRET=your-super-secret-jwt-key-minimum-32-characters-long-here
JWT_EXPIRES_IN=1h
REFRESH_TOKEN_EXPIRES_IN=7d

# === Supabase Configuration ===
# Get these from: https://app.supabase.com → Project Settings → API
SUPABASE_URL=https://your-project.supabase.co
SUPABASE_ANON_KEY=your-supabase-anon-key-here
SUPABASE_SERVICE_KEY=your-supabase-service-role-key-here

# === Frontend Configuration ===
# Development: http://localhost:5173
# Production: https://silviosuperandolimites.com.br/l2
FRONTEND_URL=https://silviosuperandolimites.com.br/l2

# === CORS Configuration ===
# Comma-separated list of allowed origins
# Development: http://localhost:5173,http://localhost:3000
# Production: https://silviosuperandolimites.com.br
ALLOWED_ORIGINS=https://silviosuperandolimites.com.br

# === Rate Limiting ===
RATE_LIMIT_WINDOW_MS=900000
RATE_LIMIT_MAX_REQUESTS=100

# ================================================
# SETUP INSTRUCTIONS:
# ================================================
#
# 1. Copy this file: cp env.example.txt .env
# 2. Generate a strong JWT_SECRET (see command above)
# 3. Get Supabase credentials from your project dashboard
# 4. Update FRONTEND_URL to your production domain
# 5. Update ALLOWED_ORIGINS to match your frontend domain(s)
#
# ================================================
# SECURITY NOTES:
# ================================================
# - JWT_SECRET must be at least 32 characters
# - SUPABASE_SERVICE_KEY has admin access - keep it secret!
# - Never commit .env file to version control
# - Use different secrets for development and production
# - FRONTEND_URL must match your actual frontend deployment
# - ALLOWED_ORIGINS controls CORS (be specific, don't use *)
